  canvas {

    background: #222;

    display: block;

    margin: 0 auto;

  }

</style>

<body>

<canvas id="gameCanvas" width="800" height="500"></canvas>

<script>

const canvas = document.getElementById('gameCanvas');

const ctx = canvas.getContext('2d');


const player = {

  x: 50,

  y: 0,

  width: 30,

  height: 50,

  color: 'lime',

  vy: 0,

  speed: 5,

  jumpStrength: 15,

  onGround: false

};


const gravity = 0.8;

const keys = {};


const platforms = [

  {x: 0, y: 450, width: 800, height: 50, color: 'brown'},

  {x: 200, y: 350, width: 120, height: 20, color: 'brown'},

  {x: 400, y: 250, width: 150, height: 20, color: 'brown'},

  {x: 600, y: 150, width: 100, height: 20, color: 'brown'}

];


const coins = [

  {x: 220, y: 330, size: 10, collected: false},

  {x: 430, y: 230, size: 10, collected: false},

  {x: 620, y: 130, size: 10, collected: false}

];


let score = 0;


document.addEventListener('keydown', (e) => {

  keys[e.key] = true;

});

document.addEventListener('keyup', (e) => {

  keys[e.key] = false;

});


function checkCollision(rect, platform) {

  return rect.x < platform.x + platform.width &&

         rect.x + rect.width > platform.x &&

         rect.y < platform.y + platform.height &&

         rect.y + rect.height > platform.y;

}


function gameLoop() {

  if (keys['ArrowLeft']) player.x -= player.speed;

  if (keys['ArrowRight']) player.x += player.speed;


  if (keys[' '] && player.onGround) {

    player.vy = -player.jumpStrength;

    player.onGround = false;

  }


  player.vy += gravity;

  player.y += player.vy;


  player.onGround = false; 

  for (const plat of platforms) {

    if (checkCollision(player, plat)) {

      player.y = plat.y - player.height;

      player.vy = 0;

      player.onGround = true;

    }

  }


  if (player.x < 0) player.x = 0;

  if (player.x + player.width > canvas.width)

    player.x = canvas.width - player.width;


  if (player.y + player.height > canvas.height) {

    player.y = canvas.height - player.height;

    player.vy = 0;

    player.onGround = true;

  }


  for (const coin of coins) {

    if (!coin.collected &&

        player.x < coin.x + coin.size &&

        player.x + player.width > coin.x &&

        player.y < coin.y + coin.size &&

        player.y + player.height > coin.y) {

      coin.collected = true;

      score++;

    }

  }


  ctx.clearRect(0, 0, canvas.width, canvas.height);


  for (const plat of platforms) {

    ctx.fillStyle = plat.color;

    ctx.fillRect(plat.x, plat.y, plat.width, plat.height);

  }


  for (const coin of coins) {

    if (!coin.collected) {

      ctx.fillStyle = 'gold';

      ctx.beginPath();

      ctx.arc(coin.x + coin.size / 2, coin.y + coin.size / 2, coin.size, 0, Math.PI * 2);

      ctx.fill();

    }

  }


  ctx.fillStyle = player.color;

  ctx.fillRect(player.x, player.y, player.width, player.height);


  ctx.fillStyle = 'white';

  ctx.font = '20px Arial';

  ctx.fillText('Собрано монет: ' + score, 10, 30);


  requestAnimationFrame(gameLoop);

}


gameLoop();

</script>

</body>
